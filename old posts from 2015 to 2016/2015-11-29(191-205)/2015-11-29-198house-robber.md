int rob(vector<int>& nums) {
  int n = nums.size();
  int ans = 0;
  vector <int> f(n, 0);

  for (int i = 0; i < n; i++) {
    for (int j = 0; j < i-1; j++) {
      f[i] = max(f[i], f[j]);
    }
    f[i] += nums[i];
    ans = max(ans, f[i]);
  }

  return ans;
}
