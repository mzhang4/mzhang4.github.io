{% highlight cpp %}

typedef long long LL;

double myPow(double x, int n) {
  LL nl = (LL)n;
  return myPow(x, nl);
}

double myPow(double x, LL n) {
  if (n < 0) return 1/myPow(x, -n);
  if (n == 0) return 1;

  double tmp = myPow(x, n/2);
  if (n & 0x1)
    return tmp * tmp * x;
  return tmp * tmp; 
}
