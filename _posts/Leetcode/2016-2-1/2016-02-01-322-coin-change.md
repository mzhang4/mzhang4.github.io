{% highlight cpp %}

class Solution {
public:
  int coinChange(vector<int>& coins, int amount) {
    vector <int> f(amount+1, -1);
    f[0] = 0;
    for (int i = 0; i < amount; i++) {
      if (f[i] == -1) continue;
      for (int j = 0; j < coins.size(); j++) {
        int s = coins[j] + i;
        if (s > amount) continue;
        if (f[s] == -1) f[s] = f[i]+1;
        else f[s] = min(f[s], f[i]+1);
      }
    }

    return f[amount];
  }
};
