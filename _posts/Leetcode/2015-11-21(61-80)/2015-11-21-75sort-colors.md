couting sort, 头放0，尾房2, 注意对i的维护

{% highlight cpp %}

void sortColors(vector<int>& nums) {
  int ind0 = 0; int ind2 = (int)nums.size() - 1;
  for (int i = 0; i <= ind2 && ind2>=0 && ind0 < nums.size(); i = max(i, ind0)) {
    if (nums[i] == 0) {
      swap(nums[ind0], nums[i]);
      ind0++;
    }
    else if (nums[i] == 2) {
      swap(nums[ind2], nums[i]);
      ind2--;
    }
    else i++;
  }
}
